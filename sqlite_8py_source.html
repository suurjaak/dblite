79  elif col.count("?") == argcount(val):

356  Transaction context manager, breakable by raising Rollback.
357 
358  Note that in SQLite, a single connection has one shared transaction state,
359  so it is highly recommended to use exclusive Transaction instances for any action queries,
360  as concurrent transactions can interfere with one another otherwise.

363  def __init__(self, db, commit=True, exclusive=True, **__):
364  """
365  Creates a new transaction.
366 
367  @param db Database instance
368  @param commit whether transaction commits automatically at exiting with-block
369  @param exclusive whether entering a with-block is exclusive over other
370  Transaction instances on this Database
371  """
372  self._db = db
373  self._exitcommit = commit

375  self._exclusive = True if exclusive is None else exclusive
376  self._closed = False
377  self._cursor = None

379  def __enter__(self):
380  """Context manager entry, opens cursor, returns Transaction object."""
381  if self._closed: raise RuntimeError("Transaction already closed")
382 
383  if self._exclusive: Database.MUTEX[self._db].acquire()
384  try: not self._cursor and self._make_cursor()
385  except Exception:
386  if self._exclusive: Database.MUTEX[self._db].release()
387  raise
388  self._enterstack += 1

391  def __exit__(self, exc_type, exc_val, exc_trace):
392  """Context manager exit, closes cursor, commits or rolls back as specified on creation."""
393  depth = self._enterstack = self._enterstack - 1
394  try:
395  if self._cursor:
396  self.commit() if self._exitcommit and exc_type is None else self.rollback()
397  return exc_type in (None, api.Rollback) # Do not propagate raised Rollback
398  finally:
399  if depth < 1:
400  self._cursor = None
401  self._closed = True
402  self._db._notify(self)
