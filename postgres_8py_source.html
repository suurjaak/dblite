77  def cast(col, val):
78  """Returns column value cast to correct type for use in psycopg."""
79  self._load_schema()
80  field = table in self._structure and self._structure[table]["fields"].get(col)

89  def parse_members(i, col, op, val):
90  """Returns (col, op, val, argkey)."""
91  key = "%sW%s" % (re.sub(r"\W+", "_", col), i)
92  if "EXPR" == col.upper():
93  # ("EXPR", ("SQL", val))
94  col, op, val, key = val[0], "EXPR", val[1], "EXPRW%s" % i
95  elif col.count("?") == argcount(val):
96  # ("any SQL with ? placeholders", val)
97  op, val, key = "EXPR", listify(val), "EXPRW%s" % i
98  elif isinstance(val, (list, tuple)) and len(val) == 2 \
99  and isinstance(val[0], string_types):
100  tmp = val[0].strip().upper()
101  if tmp in self.OPS: # ("col", ("binary op like >=", val))
102  op, val = tmp, val[1]
103  elif val[0].count("?") == argcount(val[1]):
104  # ("col", ("SQL with ? placeholders", val))
105  col, val, op = "%s = %s" % (col, val[0]), listify(val[1]), "EXPR"
106  if op in ("IN", "NOT IN") and not val: # IN -> ANY, to avoid error on empty array
107  col = "%s%s = ANY('{}')" % ("" if "IN" == op else "NOT ", col)
108  op = "EXPR"
109  return col, op, val, key
110  def argcount(x) : return len(x) if isinstance(x, (list, set, tuple)) else 1
111  def listify(x) : return x if isinstance(x, (list, tuple)) else [x]
112  def strlistify(x): return [text_type(y) for y in listify(x) if y not in ("", None)]
113 
114  action = action.upper()
115  cols = cols if isinstance(cols, string_types) else ", ".join(strlistify(cols)) or "*"
116  where = [where] if isinstance(where, string_types) else where
117  group = group if isinstance(group, string_types) else ", ".join(strlistify(group))
118  order = [order] if isinstance(order, string_types) else order
119  order = [order] if isinstance(order, (list, tuple)) \
120  and len(order) == 2 and isinstance(order[1], bool) else order
121  limit = [limit] if isinstance(limit, string_types + integer_types) else limit
122  values = values if not isinstance(values, dict) else values.items()
123  where = where if not isinstance(where, dict) else where.items()
124  sql = "SELECT %s FROM %s" % (cols, table) if "SELECT" == action else ""
125  sql = "DELETE FROM %s" % (table) if "DELETE" == action else sql
126  sql = "INSERT INTO %s" % (table) if "INSERT" == action else sql
127  sql = "UPDATE %s" % (table) if "UPDATE" == action else sql
128  args = {}
129 
130  if "INSERT" == action:
131  keys = ["%sI%s" % (re.sub(r"\W+", "_", k), i) for i, (k, _) in enumerate(values)]
132  args.update((n, cast(k, v)) for n, (k, v) in zip(keys, values))
133  cols, vals = ", ".join(k for k, _ in values), ", ".join("%%(%s)s" % n for n in keys)
134  sql += " (%s) VALUES (%s)" % (cols, vals)
135  self._load_schema()
136  if table in self._structure and self._structure[table].get("key"):
137  sql += " RETURNING %s AS id" % (self._structure[table]["key"])
138  if "UPDATE" == action:
139  sql += " SET "
140  for i, (col, val) in enumerate(values):
141  key = "%sU%s" % (re.sub(r"\W+", "_", col), i)
142  sql += (", " if i else "") + "%s = %%(%s)s" % (col, key)
143  args[key] = cast(col, val)
144  if where:
145  sql += " WHERE "
146  for i, clause in enumerate(where):
147  if isinstance(clause, string_types): # "raw SQL with no arguments"
148  clause = (clause, )
149 
150  if len(clause) == 1: # ("raw SQL with no arguments", )
151  col, op, val, key = clause[0], "EXPR", [], None
152  elif len(clause) == 2: # ("col", val) or ("col", ("op" or "expr with ?", val))
153  col, op, val, key = parse_members(i, clause[0], "=", clause[1])
154  else: # ("col", "op" or "expr with ?", val)
155  col, op, val, key = parse_members(i, *clause)
156 
157  if "EXPR" == op:
158  for j in range(col.count("?")):
159  col = col.replace("?", "%%(%s_%s)s" % (key, j), 1)
160  args["%s_%s" % (key, j)] = cast(None, val[j])
161  sql += (" AND " if i else "") + "(%s)" % col
162  elif val is None:
163  op = {"=": "IS", "!=": "IS NOT", "<>": "IS NOT"}.get(op, op)
164  sql += (" AND " if i else "") + "%s %s NULL" % (col, op)
165  else:
166  args[key] = cast(col, val)
167  sql += (" AND " if i else "") + "%s %s %%(%s)s" % (col, op, key)
168  if group:
169  sql += " GROUP BY " + group
170  if order:
171  sql += " ORDER BY "
172  for i, col in enumerate(listify(order)):
173  name = col if isinstance(col, string_types) else \
174  text_type(col[0] if isinstance(col, (list, tuple)) else col)
175  sort = col[1] if name != col and isinstance(col, (list, tuple)) and len(col) > 1 \
176  else ""
177  if not isinstance(sort, string_types): sort = "DESC" if sort else ""
178  sql += (", " if i else "") + name + (" " if sort else "") + sort
179  if limit:
180  limit = [None if isinstance(v, integer_types) and v < 0 else v for v in limit]
181  for k, v in zip(("limit", "offset"), limit):
182  if v is None: continue # for k, v
183  sql += " %s %%(%s)s" % (k.upper(), k)
184  args[k] = v
185 
186  return sql, args

362  @contextmanager
363  def make_cursor(self, commit=False, autocommit=False, schema=None, lazy=False, itersize=None):
364  """
365  Context manager for psycopg connection cursor.
366  Creates a new cursor on an unused connection and closes it when exiting
367  context, committing changes if specified.
368 
369  @param commit commit at the end on success
370  @param autocommit connection autocommit mode
371  @param schema name of Postgres schema to use, if not using default `"public"`
372  @param lazy if true, returns a named server-side cursor that fetches rows
373  iteratively in batches; only supports making a single query
374  @param itersize batch size in rows for server-side cursor
375  @return psycopg2.extras.RealDictCursor
376  """
377  connection = self.POOLS[self].getconn()
378  try:
379  connection.autocommit = autocommit
380  cursor, namedcursor = None, None
381  if "public" == schema: schema = None # Default, no need to set
382 
383  # If using schema, schema tables are queried first, fallback to public.
384  # Need two cursors if schema+lazy, as named cursor only does one query.
385  if schema or not lazy: cursor = connection.cursor()
386  if schema: cursor.execute('SET search_path TO %s,public' % quote(schema))
387  if lazy: namedcursor = connection.cursor("name_%s" % id(connection))
388  if lazy and itersize is not None: namedcursor.itersize = itersize
389 
390  try:
391  yield namedcursor or cursor
392  if commit: connection.commit()
393  except GeneratorExit: pass # Caller consumed nothing
394  except Exception as e:
395  if not isinstance(e, api.Rollback):
396  logger.exception("SQL error on %s:", (namedcursor or cursor).query)
397  raise
398  finally:
399  connection.rollback() # If not already committed, must rollback here
400  try: namedcursor and namedcursor.close()
401  except Exception: pass
402  if schema: # Restore default search path on this connection
403  cursor.execute("SET search_path TO public")
404  connection.commit()
405  if cursor: cursor.close()
406  finally: self.POOLS[self].putconn(connection)

444  Supports server-side cursors; those can only be used for making a single query.

450  def __init__(self, db, commit=True, exclusive=False,
451  schema=None, lazy=False, itersize=None, **__):
452  """
453  Creates a transaction context manager.
454 
455  Context is breakable by raising Rollback.
456 
457  @param db Database instance
458  @param commit whether transaction commits automatically at exiting with-block
459  @param exclusive whether entering a with-block is exclusive over other
460  Transaction instances Database
461  @param schema search_path to use in this transaction
462  @param lazy if true, uses a server-side cursor to fetch results from server
463  iteratively in batches instead of all at once,
464  supports one single query only
465  @param itersize batch size for server-side cursor (defaults to 2000 rows)
466  """
467  self._db = db
468  self._lazy = lazy
469  self._cursor = None
470  self._cursorctx = db.make_cursor(commit, schema=schema, lazy=lazy, itersize=itersize)
471  self._exclusive = exclusive
472  self._exitcommit = commit
473  self._enterstack = 0 # Number of levels the transaction context is nested at
474  self._structure = None # Database schema as {table or view name: {"fields": {..}, ..}}
475 
476  def __enter__(self):
477  """Context manager entry, opens cursor, returns Transaction object."""
478  if self.closed: raise RuntimeError("Transaction already closed")
479 
480  if self._exclusive: Database.MUTEX[self._db].acquire()
481  try:
482  if not self._cursor: self._cursor = self._cursorctx.__enter__()
483  self._enterstack += 1
484  return self
485  except Exception:
486  if self._exclusive: Database.MUTEX[self._db].release()
487  raise
488 
489  def __exit__(self, exc_type, exc_val, exc_trace):
490  """Context manager exit, closes cursor, commits or rolls back as specified on creation."""
491  depth = self._enterstack = self._enterstack - 1
492  try:
493  if self._cursor and depth < 1: # Last level: close properly
494  self._cursorctx.__exit__(exc_type, exc_val, exc_trace)
495  elif self._cursor: # Still some depth: intermediary commit/rollback
496  self.commit() if self._exitcommit and exc_type is None else self.rollback()
497  return exc_type in (None, api.Rollback)
498  finally:
499  if depth < 1:
500  self._cursor = None
501  self._cursorctx = None
502  self._db._notify(self)
503  if self._exclusive: Database.MUTEX[self._db].release()

505  def close(self, commit=None):
506  """
507  Closes the transaction, performing commit or rollback as specified,
508  and releases database connection back to connection pool.
509  Required if not using transaction as context manager in a with-block.
510 
511  @param commit `True` for explicit commit, `False` for explicit rollback,
512  `None` defaults to `commit` flag from creation
513  """
514  if not self._cursor:
515  self._db._notify(self)
516  return
517  if commit is False: self.rollback()
518  elif commit: self.commit()
519  try: self._cursorctx.__exit__(None, None, None)
520  finally:
521  self._cursor = None
522  self._cursorctx = None
523  self._db._notify(self)

602  """Returns Postgres connection options as URL, like `"postgresql://host/dbname"`."""
603  BASICS = collections.OrderedDict([("user", ""), ("password", ":"), ("host", ""),
604  ("port", ":"), ("dbname", "/")])
605  result, creds = "", False
606  if isinstance(opts, string_types):
607  opts = psycopg2.extensions.parse_dsn(opts)
608  for i, (k, prefix) in enumerate(BASICS.items()):
609  if creds and i > 1: result, creds = result + "@", False # Either user or password set
610  if opts.get(k) is not None:
611  result, creds = result + prefix + "%%(%s)s" % k, (i < 2)
612  result %= {k : urllib_parse.quote(text_type(opts[k])) for k in opts}
613  if any(k not in BASICS for k in opts):
614  result += "/" if opts.get("dbname") is None else ""
615  result += "?" + urllib_parse.urlencode({k: opts[k] for k in opts if k not in BASICS})
616  return "postgresql://" + result

619 def query_schema(queryable, keys=False, views=False, inheritance=False):
620  """
621  Returns database table structure populated from given database.
622 
623  @param queryable Database or Transaction instance
624  @param views whether to include views
625  @param keys whether to include primary and foreign key information
626  @param inheritance whether to include parent-child table information
627  and populate inherited foreign keys
628  @return ```{table or view name: {
629  "fields": OrderedDict({
630  column name: {
631  "name": column name,
632  "type": column type name,
633  ?"pk": True,
634  ?"fk": foreign table name,
635  }
636  }),
637  ?"key": primary key column name,
638  ?"parent": parent table name,
639  ?"children": [child table name, ],
640  "type": "table" or "view",
641  }
642  }```
643  """
644  result = {}
645 
646  # Retrieve column names
647  for v in queryable.fetchall("information_schema.columns", table_schema="public",
648  order="table_name, ordinal_position"):
649  t, c, d = v["table_name"], v["column_name"], v["data_type"]
650  if t not in result: result[t] = {"type": "table",
651  "fields": collections.OrderedDict()}
652  result[t]["fields"][c] = {"name": c, "type": d.lower()}
653 
654  # Retrieve primary and foreign keys
655  for v in queryable.fetchall(
656  "information_schema.table_constraints tc "
657  "JOIN information_schema.key_column_usage kcu "
658  "ON tc.constraint_name = kcu.constraint_name "
659  "JOIN information_schema.constraint_column_usage ccu "
660  "ON ccu.constraint_name = tc.constraint_name ",
661  cols="DISTINCT tc.table_name, kcu.column_name, tc.constraint_type, "
662  "ccu.table_name AS table_name2", where={"tc.table_schema": "public"}
663  ) if keys else ():
664  t, c, t2 = v["table_name"], v["column_name"], v["table_name2"]
665  if "PRIMARY KEY" == v["constraint_type"]:
666  result[t]["fields"][c]["pk"], result[t]["key"] = True, c
667  else: result[t]["fields"][c]["fk"] = t2
668 
669  # Retrieve inheritance information, copy foreign key flags from parent
670  for v in queryable.fetchall(
671  "information_schema.pg_inherits i JOIN information_schema.pg_class c ON inhrelid=c.oid "
672  "JOIN information_schema.pg_class p ON inhparent = p.oid "
673  "JOIN information_schema.pg_namespace pn ON pn.oid = p.relnamespace "
674  "JOIN information_schema.pg_namespace cn "
675  "ON cn.oid = c.relnamespace AND cn.nspname = pn.nspname",
676  cols="c.relname AS child, p.relname AS parent",
677  where={"pn.nspname": "public"}
678  ) if inheritance else ():
679  result[v["parent"]].setdefault("children", []).append(v["child"])
680  result[v["child"]]["parent"] = v["parent"]
681  for f, opts in result[v["parent"]]["fields"].items() if keys else ():
682  if not opts.get("fk"): continue # for f, opts
683  result[v["child"]]["fields"][f]["fk"] = opts["fk"]
684 
685  # Retrieve view column names
686  for v in queryable.fetchall(
687  "information_schema.pg_attribute a "
688  "JOIN information_schema.pg_class c ON a.attrelid = c.oid "
689  "JOIN information_schema.pg_namespace s ON c.relnamespace = s.oid "
690  "JOIN information_schema.pg_type t ON a.atttypid = t.oid "
691  "JOIN information_schema.pg_proc p ON t.typname = p.proname ",
692  cols="DISTINCT c.relname, a.attname, pg_get_function_result(p.oid) AS data_type",
693  where={"a.attnum": (">", 0), "a.attisdropped": False,
694  "s.nspname": "public", "c.relkind": ("IN", ("v", "m"))}
695  ) if views else ():
696  t, c, d = v["relname"], v["attname"], v["data_type"]
697  if t not in result: result[t] = {"type": "view",
698  "fields": collections.OrderedDict()}
699  result[t]["fields"][c] = {"name": c, "type": d.lower()}
700 
701  return result

704 def quote(value, force=False):
705  """
706  Returns identifier in quotes and proper-escaped for queries,
707  if value needs quoting (has non-alphanumerics, starts with number, or is reserved).
708 
709  @param value the value to quote, returned as-is if not string
710  @param force whether to quote value even if not required
711  """
712  if not isinstance(value, string_types):
713  return value
714  RGX_INVALID, RGX_UNICODE = r"(^[\W\d])|(?=\W)", r"[^\x01-\x7E]"
715  result = value.decode() if isinstance(value, binary_type) else value
716  if force or result.upper() in RESERVED_KEYWORDS or re.search(RGX_INVALID, result):
717  if re.search(RGX_UNICODE, value): # Convert to Unicode escape U&"\+ABCDEF"
718  result = result.replace("\\", r"\\").replace('"', '""')
719  result = 'U&"%s"' % re.sub(RGX_UNICODE, lambda m: r"\+%06X" % ord(m.group(0)), value)
720  else:
721  result = '"%s"' % result.replace('"', '""')
722  return result
